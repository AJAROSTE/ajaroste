domain= pcde birbirne görmesini sağlayan nümerlik sisteminin daha kolaylaştırılmış (rahatça girilebilmesi için kelimelere ifade edilir)

hosting= siteyi barındırmak için kullanılan yer

ol= sıralı cümle oluşturmak için kullanılıyor harf rakam ve ay roma rakamı ile listelenen ol <li> etiketi ile yapılır

type=hangi şekilde yazacağımıza karar veririz

start=her zaman sayı olması gerekir

ul= madde işaretleri şeklinde listeleme yapmak için kullanılır (disc=içi dolu yuvarlak circle=içi boş yuvarlak square içi dolu kare)

li=liste oluştumak için elemanları belirtmekte kullanılan etikettir

colspan= satırları birleştirir

rowspan= sütünları birleştirir

B=kalın yazmak içn

U=altı çizgili yazmak içi

İ=italik yazı tipi

br= alt satıra geçmek için

p= paragraf boşluğu bırakmak için

hr= yatay çizgi çekmek için kullanılan etikettir (size =çizilen çizginin kalınlığını belirlemek için kullanılır width= çizginin genişliğini ayarlar align = çizginin sayfada konumunu belirler)

table= satır ve sütünlardan oluşur

TR=tobloda satır oluşturmak için kullanılır

td=tabloda sütun oluşturmak için kullanılır

border= hücrenin etrafındaki çerçevenin kalınlığını ayarlar
